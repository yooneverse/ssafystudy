# 문제 3: 버블 소트

arr = [5, 1, 4, 2, 8]

n = len(arr)
for i in range(n - 1, 0, -1):
    # for i in range(n) 을 수정했습니다.
    # 이유: 버블 소트는 i를 1회 돌 때마다 배열의 마지막 1칸이 줄어듭니다.
    # 그리고, 각 원소 비교를 위해 현재 j와 j + 1의 요소를 비교해야합니다.
    # 그러므로 i는 배열이 한 칸씩 줄어들 수 있도록 최대 길이(n)부터 반복하되,
    # j 반복문에서 배열의 범위를 넘는 일이 발생하지 않도록 n - 1부터 시작해야 합니다.

    for j in range(i):
        if arr[j] > arr[j + 1]:
            # arr[j] < arr[j + 1]을 수정했습니다
            # 이유: 오름차순으로 정열하기 위해선 더 큰 원소가 오른쪽으로 이동해야합니다.
            # 고치기 전의 문장은 더 큰 원소가 왼쪽으로 이동하게 되어 내림차순으로 정렬됩니다.

            arr[j], arr[j + 1] = arr[j + 1], arr[j]
            # arr[j], arr[j + 1] == arr[j + 1], arr[j]을 수정했습니다.
            # 이유: '==' 연산자는 각 원소의 값이 동일한지 비교하는 연산자입니다.
            # 현재 위치에는 j번째 원소가 j + 1번째 원소보다 클 경우,
            # 각 위치에 값을 다시 할당하는 연산을 해야하므로 '=' 연산자가 들어가야합니다.
